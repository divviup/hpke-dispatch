Expand description

Types which have an ASN.1 Tag.

§Example

use der::{Tag, Tagged};

/// Struct, which Tag depends on data
struct MyOctetOrBitString(bool);

impl Tagged for MyOctetOrBitString {
    fn tag(&self) -> Tag {
        if self.0 {
            Tag::OctetString
        } else {
            Tag::BitString
        }
    }
}
