89/// A sequence counter. This is set to `u64` instead of the true nonce size of an AEAD for two
90/// reasons:
91///
92/// 1. No algorithm that would appear in HPKE would require nonce sizes less than `u64`.
93/// 2. It is just about physically impossible to encrypt 2^64 messages in sequence. If a computer
94///    computes 1 encryption every nanosecond, it would take over 584 years to run out of nonces.
95///    Notably, unlike randomized nonces, counting in sequence doesn't parallelize, so we don't
96///    have to imagine amortizing this computation across multiple computers. In conclusion, 64
97///    bits should be enough for anybody.
98#[derive(Clone, Default, Zeroize, ZeroizeOnDrop)]
99struct Seq(u64);
100
101// RFC 9180 §5.2
102// def Context<ROLE>.IncrementSeq():
103//   if self.seq >= (1 << (8*Nn)) - 1:
104//     raise MessageLimitReachedError
105//   self.seq += 1
106
107/// Increments the sequence counter. Returns `None` on overflow.
108fn increment_seq(seq: &Seq) -> Option<Seq> {
109    // Try to add 1
110    seq.0.checked_add(1).map(Seq)
111}
112
113// RFC 9180 §5.2
114// def Context<ROLE>.ComputeNonce(seq):
115//   seq_bytes = I2OSP(seq, Nn)
116//   return xor(self.base_nonce, seq_bytes)
117
118/// Derives a nonce from the base nonce and a "sequence number". The sequence number is treated as
119/// a big-endian integer with length equal to the nonce length.
120fn mix_nonce<A: Aead>(base_nonce: &AeadNonce<A>, seq: &Seq) -> AeadNonce<A> {
121    // Write `seq` in big-endian order into a byte buffer that's the size of a nonce
122    let mut seq_buf = AeadNonce::<A>::default();
123    // We just write to the last seq_size bytes. This is necessary because our AEAD nonces (>= 96
124    // bits) are always bigger than the sequence buffer (64 bits). We write to the last 64 bits
125    // because this is a big-endian number.
126    let seq_size = core::mem::size_of::<Seq>();
127    let nonce_size = base_nonce.0.len();
128    write_u64_be(&mut seq_buf.0[nonce_size - seq_size..], seq.0);
129
130    // XOR the base nonce bytes with the sequence bytes
131    let new_nonce_iter = base_nonce
132        .0
133        .iter()
134        .zip(seq_buf.0.iter())
135        .map(|(nonce_byte, seq_byte)| nonce_byte ^ seq_byte);
136
137    // This cannot fail, as new_nonce_iter is exactly the length of base_nonce, which is itself an
138    // AeadNonce<A>
139    AeadNonce(Array::try_from_iter(new_nonce_iter).unwrap())
140}

152impl<A: Aead> Serializable for AeadTag<A> {
153    type OutputSize = <A::AeadImpl as BaseAeadCore>::TagSize;
154
155    // Pass to underlying to_bytes() impl
156    fn write_exact(&self, buf: &mut [u8]) {
157        // Check the length is correct and panic if not
158        enforce_outbuf_len::<Self>(buf);
159
160        buf.copy_from_slice(&self.0);
161    }
162}

164impl<A: Aead> Deserializable for AeadTag<A> {
165    fn from_bytes(encoded: &[u8]) -> Result<Self, HpkeError> {
166        enforce_equal_len(Self::size(), encoded.len())?;
167
168        // Copy to a fixed-size array
169        let mut arr = <Array<u8, Self::OutputSize> as Default>::default();
170        arr.copy_from_slice(encoded);
171        Ok(AeadTag(arr))
172    }

175/// The HPKE encryption context. This is what you use to `seal` plaintexts and `open` ciphertexts.

213        key: &AeadKey<A>,
214        base_nonce: AeadNonce<A>,

218        AeadCtx {
219            overflowed: false,
220            encryptor: <A::AeadImpl as aead::KeyInit>::new(&key.0),
221            base_nonce,
222            exporter_secret,
223            seq: <Seq as Default>::default(),
224            src_kem: PhantomData,
225            suite_id,
226        }
227    }

229    // RFC 9180 §5.3
230    // def Context.Export(exporter_context, L):
231    //   return LabeledExpand(self.exporter_secret, "sec",
232    //                        exporter_context, L)
233
234    /// Fills a given buffer with secret bytes derived from this encryption context. This value
235    /// does not depend on sequence number, so it is constant for the lifetime of this context.
236    ///
237    /// Return Value
238    /// ============
239    /// Returns `Err(HpkeError::KdfOutputTooLong)` if `out_buf.len()` ≥ 2¹⁶. Otherwise returns

244            &self.suite_id,

246            out_buf,
247        )
248    }

269impl<A: Aead, Kdf: KdfTrait, Kem: KemTrait> AeadCtxR<A, Kdf, Kem> {
270    // RFC 9180 §5.2
271    // def ContextR.Open(aad, ct):
272    //   pt = Open(self.key, self.ComputeNonce(self.seq), aad, ct)
273    //   if pt == OpenError:
274    //     raise OpenError
275    //   self.IncrementSeq()
276    //   return pt
277
278    /// Decrypts the data in `buffer`, taking the authentication tag as a separate input
279    ///
280    /// Return Value
281    /// ============
282    /// Returns `Ok(())` on success. If this context has been used for so many encryptions that the
283    /// sequence number overflowed, returns `Err(HpkeError::MessageLimitReached)`. If this happens,
284    /// `ciphertext` will be unmodified. If the tag fails to validate, returns
285    /// `Err(HpkeError::OpenError)`. If this happens, `ciphertext` is in an undefined state.
286    pub fn open_inout_detached(
287        &mut self,
288        buffer: InOutBuf<'_, '_, u8>,
289        aad: &[u8],
290        tag: &AeadTag<A>,
291    ) -> Result<(), HpkeError> {
292        if self.0.overflowed {
293            // If the sequence counter overflowed, we've been used for too long. Shut down.
294            Err(HpkeError::MessageLimitReached)
295        } else {
296            // Compute the nonce and do the encryption in place
297            let nonce = mix_nonce::<A>(&self.0.base_nonce, &self.0.seq);
298            let decrypt_res = self
299                .0
300                .encryptor
301                .decrypt_inout_detached(&nonce.0, aad, buffer, &tag.0);
302
303            if decrypt_res.is_err() {
304                // Opening failed due to a bad tag
305                return Err(HpkeError::OpenError);
306            }
307
308            // Opening was a success. Try to increment the sequence counter. If it fails, this was
309            // our last decryption.
310            match increment_seq(&self.0.seq) {
311                Some(new_seq) => self.0.seq = new_seq,
312                None => self.0.overflowed = true,
313            }
314
315            Ok(())
316        }
317    }
318
319    /// Opens the given ciphertext and returns a plaintext
320    ///
321    /// Return Value
322    /// ============
323    /// Returns `Ok(())` on success. If this context has been used for so many encryptions that the
324    /// sequence number overflowed, returns `Err(HpkeError::MessageLimitReached)`. If the tag fails
325    /// to validate, returns `Err(HpkeError::OpenError)`.
326    #[cfg(feature = "alloc")]
327    pub fn open(&mut self, ciphertext: &[u8], aad: &[u8]) -> Result<crate::Vec<u8>, HpkeError> {
328        // Make sure the auth'd ciphertext is long enough to contain a tag. If it isn't, it's
329        // certainly not valid.
330
331        let tag_len = AeadTag::<A>::size();
332        let msg_len = ciphertext
333            .len()
334            .checked_sub(tag_len)
335            .ok_or(HpkeError::OpenError)?;
336
337        // Now deconstruct the auth'd ciphertext
338        let (ciphertext, tag_slice) = ciphertext.split_at(msg_len);
339        let tag = {
340            let mut t = <AeadTag<A> as Default>::default();
341            t.0.copy_from_slice(tag_slice);
342            t
343        };
344
345        // Decrypt and return the decrypted buffer
346        let mut outbuf = vec![0u8; msg_len];
347        // We can unwrap the inout value because outbuf and ciphertext are exactly msg_len long
348        self.open_inout_detached(
349            InOutBuf::new(ciphertext, outbuf.as_mut_slice()).unwrap(),
350            aad,
351            &tag,
352        )?;
353        Ok(outbuf)
354    }
355
356    /// Fills a given buffer with secret bytes derived from this encryption context. This value
357    /// does not depend on sequence number, so it is constant for the lifetime of this context.
358    ///
359    /// Return Value
360    /// ============
361    /// Returns `Err(HpkeError::KdfOutputTooLong)` if `out_buf.len()` ≥ 2¹⁶. Otherwise returns
362    /// `Ok(())`. Just don't use this function to fill massive buffers and you'll be fine.
363    pub fn export(&self, info: &[u8], out_buf: &mut [u8]) -> Result<(), HpkeError> {
364        // Pass to AeadCtx
365        self.0.export(info, out_buf)
366    }
367}

387impl<A: Aead, Kdf: KdfTrait, Kem: KemTrait> AeadCtxS<A, Kdf, Kem> {
388    // RFC 9180 §5.2
389    // def ContextS.Seal(aad, pt):
390    //   ct = Seal(self.key, self.ComputeNonce(self.seq), aad, pt)
391    //   self.IncrementSeq()
392    //   return ct
393
394    /// Encrypts the data in `buffer`, returning the resulting authentication tag
395    ///
396    /// Return Value
397    /// ============
398    /// Returns `Ok(tag)` on success.  If this context has been used for so many encryptions that
399    /// the sequence number overflowed, returns `Err(HpkeError::MessageLimitReached)`. If this
400    /// happens, `plaintext` will be unmodified. If an error happened during encryption, returns
401    /// `Err(HpkeError::SealError)`. If this happens, the contents of `plaintext` is undefined.
402    pub fn seal_inout_detached(
403        &mut self,
404        buffer: InOutBuf<'_, '_, u8>,
405        aad: &[u8],
406    ) -> Result<AeadTag<A>, HpkeError> {
407        if self.0.overflowed {
408            // If the sequence counter overflowed, we've been used for far too long. Shut down.
409            Err(HpkeError::MessageLimitReached)
410        } else {
411            // Compute the nonce and do the encryption in place
412            let nonce = mix_nonce::<A>(&self.0.base_nonce, &self.0.seq);
413            let tag = self
414                .0
415                .encryptor
416                .encrypt_inout_detached(&nonce.0, aad, buffer)
417                .map_err(|_| HpkeError::SealError)?;
418
419            // Try to increment the sequence counter. If it fails, this was our last encryption.
420            match increment_seq(&self.0.seq) {
421                Some(new_seq) => self.0.seq = new_seq,
422                None => self.0.overflowed = true,
423            }
424
425            // Return the tag
426            Ok(AeadTag(tag))
427        }
428    }
429
430    /// Seals the given plaintext and returns the ciphertext
431    ///
432    /// Return Value
433    /// ============
434    /// Returns `Ok(ciphertext)` on success.  If this context has been used for so many encryptions
435    /// that the sequence number overflowed, returns `Err(HpkeError::MessageLimitReached)`. If an
436    /// error happened during encryption, returns `Err(HpkeError::SealError)`.
437    #[cfg(feature = "alloc")]
438    pub fn seal(&mut self, plaintext: &[u8], aad: &[u8]) -> Result<crate::Vec<u8>, HpkeError> {
439        let msg_len = plaintext.len();
440        let tag_len = AeadTag::<A>::size();
441
442        // Make a buffer that can hold a ciphertext + tag. Copy in the plaintext
443        let mut outbuf = vec![0u8; msg_len + tag_len];
444        outbuf[..msg_len].copy_from_slice(plaintext);
445
446        // Seal with a detached tag
447        let tag = self.seal_inout_detached(
448            InOutBuf::new(plaintext, &mut outbuf[..plaintext.len()]).unwrap(),
449            aad,
450        )?;
451        // Then append the tag to the end of the buffer. The buffer is now the auth'd ciphertext
452        outbuf[msg_len..msg_len + tag_len].copy_from_slice(&tag.0);
453
454        Ok(outbuf)
455    }
456
457    /// Fills a given buffer with secret bytes derived from this encryption context. This value
458    /// does not depend on sequence number, so it is constant for the lifetime of this context.
459    ///
460    /// Return Value
461    /// ============
462    /// Returns `Err(HpkeError::KdfOutputTooLong)` if `out_buf.len()` ≥ 2¹⁶. Otherwise returns
463    /// `Ok(())`. Just don't use this function to fill massive buffers and you'll be fine.
464    pub fn export(&self, info: &[u8], out_buf: &mut [u8]) -> Result<(), HpkeError> {
465        // Pass to AeadCtx
466        self.0.export(info, out_buf)
467    }
468}

593    /// Tests that sequence overflowing causes an error. This logic is cipher-agnostic, so we don't
594    /// make the test generic over ciphers.
595    #[cfg(feature = "alloc")]
596    macro_rules! test_overflow {
597        ($test_name:ident, $kdf_ty:ty, $kem_ty:ty) => {
598            #[test]
599            fn $test_name() {
600                type Kem = $kem_ty;
601                type Kdf = $kdf_ty;
602                // Again, this test is cipher-agnostic
603                type A = ChaCha20Poly1305;
604
605                // Make a sequence number that's at the max
606                let big_seq = {
607                    let mut seq = <Seq as Default>::default();
608                    seq.0 = u64::MAX;
609                    seq
610                };
611
612                let (mut sender_ctx, mut receiver_ctx) = gen_ctx_simple_pair::<A, Kdf, Kem>();
613                sender_ctx.0.seq = big_seq.clone();
614                receiver_ctx.0.seq = big_seq.clone();
615
616                // These should support precisely one more encryption before it registers an
617                // overflow
618
619                let msg = b"draxx them sklounst";
620                let aad = b"you have to have the kebapi";
621
622                // Do one round trip and ensure it works
623                {
624                    let mut buf = msg.clone();
625
626                    // Encrypt the plaintext
627                    let ciphertext = sender_ctx.seal(&mut buf, aad).expect("seal() failed");
628
629                    // Now to decrypt on the other side
630                    let roundtrip_plaintext =
631                        receiver_ctx.open(&ciphertext, aad).expect("open() failed");
632
633                    // Make sure the output message was the same as the input message
634                    assert_eq!(msg, roundtrip_plaintext.as_slice());
635                }
636
637                // Try another round trip and ensure that we've overflowed
638                {
639                    // Try to encrypt the plaintext
640                    match sender_ctx.seal(msg, aad) {
641                        Err(HpkeError::MessageLimitReached) => {
642                            // Good, this should have overflowed
643                        }
644                        Err(e) => panic!("seal() should have overflowed. Instead got {}", e),
645                        _ => panic!("seal() should have overflowed. Instead it succeeded"),
646                    }
647
648                    // Now try to decrypt something. This isn't a valid ciphertext or tag, but the
649                    // overflow should fail before the tag check fails.
650                    let placeholder_ciphertext = [0u8; 32];
651
652                    match receiver_ctx.open(&placeholder_ciphertext, aad) {
653                        Err(HpkeError::MessageLimitReached) => {
654                            // Good, this should have overflowed
655                        }
656                        Err(e) => panic!("open() should have overflowed. Instead got {}", e),
657                        _ => panic!("open() should have overflowed. Instead it succeeded"),
658                    }
659                }
660            }
661        };
662    }

664    /// Tests that `open()` can decrypt things properly encrypted with `seal()`
665    #[cfg(feature = "alloc")]
666    macro_rules! test_ctx_correctness {
667        ($test_name:ident, $aead_ty:ty, $kdf_ty:ty, $kem_ty:ty) => {
668            #[test]
669            fn $test_name() {
670                type A = $aead_ty;
671                type Kdf = $kdf_ty;
672                type Kem = $kem_ty;
673
674                let (mut sender_ctx, mut receiver_ctx) = gen_ctx_simple_pair::<A, Kdf, Kem>();
675
676                let msg = b"Love it or leave it, you better gain way";
677                let aad = b"You better hit bull's eye, the kid don't play";
678
679                // Encrypt in place with the sender context
680                let ciphertext = sender_ctx.seal(msg, aad).expect("seal() failed");
681
682                // Make sure seal() isn't a no-op
683                assert_ne!(&ciphertext, msg);
684
685                // Decrypt with the receiver context
686                let decrypted = receiver_ctx.open(&ciphertext, aad).expect("open() failed");
687                assert_eq!(&decrypted, msg);
688
689                // Now try sending an invalid message followed by a valid message. The valid
690                // message should decrypt correctly
691                let invalid_ciphertext = [0x00; 32];
692                assert!(receiver_ctx.open(&invalid_ciphertext, aad).is_err());
693
694                // Now make sure a round trip succeeds
695                let ciphertext = sender_ctx.seal(msg, aad).expect("second seal() failed");
696
697                // Decrypt with the receiver context
698                let decrypted = receiver_ctx
699                    .open(&ciphertext, aad)
700                    .expect("second open() failed");
701                assert_eq!(&decrypted, msg);
702            }
703        };
704    }
