75/// A sequence counter. This is set to `u64` instead of the true nonce size of an AEAD for two
76/// reasons:
77///
78/// 1. No algorithm that would appear in HPKE would require nonce sizes less than `u64`.
79/// 2. It is just about physically impossible to encrypt 2^64 messages in sequence. If a computer
80///    computes 1 encryption every nanosecond, it would take over 584 years to run out of nonces.
81///    Notably, unlike randomized nonces, counting in sequence doesn't parallelize, so we don't
82///    have to imagine amortizing this computation across multiple computers. In conclusion, 64
83///    bits should be enough for anybody.
84#[derive(Clone, Default, Zeroize)]
85#[zeroize(drop)]
86struct Seq(u64);
87
88// RFC 9180 §5.2
89// def Context<ROLE>.IncrementSeq():
90//   if self.seq >= (1 << (8*Nn)) - 1:
91//     raise MessageLimitReachedError
92//   self.seq += 1
93
94/// Increments the sequence counter. Returns `None` on overflow.
95fn increment_seq(seq: &Seq) -> Option<Seq> {
96    // Try to add 1
97    seq.0.checked_add(1).map(Seq)
98}
99
100// RFC 9180 §5.2
101// def Context<ROLE>.ComputeNonce(seq):
102//   seq_bytes = I2OSP(seq, Nn)
103//   return xor(self.base_nonce, seq_bytes)
104
105/// Derives a nonce from the base nonce and a "sequence number". The sequence number is treated as
106/// a big-endian integer with length equal to the nonce length.
107fn mix_nonce<A: Aead>(base_nonce: &AeadNonce<A>, seq: &Seq) -> AeadNonce<A> {
108    // Write `seq` in big-endian order into a byte buffer that's the size of a nonce
109    let mut seq_buf = AeadNonce::<A>::default();
110    // We just write to the last seq_size bytes. This is necessary because our AEAD nonces (>= 96
111    // bits) are always bigger than the sequence buffer (64 bits). We write to the last 64 bits
112    // because this is a big-endian number.
113    let seq_size = core::mem::size_of::<Seq>();
114    let nonce_size = base_nonce.0.len();
115    write_u64_be(&mut seq_buf.0[nonce_size - seq_size..], seq.0);
116
117    // XOR the base nonce bytes with the sequence bytes
118    let new_nonce_iter = base_nonce
119        .0
120        .iter()
121        .zip(seq_buf.0.iter())
122        .map(|(nonce_byte, seq_byte)| nonce_byte ^ seq_byte);
123
124    // This cannot fail, as the length of AeadNonce<A> is precisely the length of Seq
125    AeadNonce(GenericArray::from_exact_iter(new_nonce_iter).unwrap())
126}

161/// The HPKE encryption context. This is what you use to `seal` plaintexts and `open` ciphertexts.
162pub(crate) struct AeadCtx<A: Aead, Kdf: KdfTrait, Kem: KemTrait> {
163    /// Records whether the nonce sequence counter has overflowed
164    overflowed: bool,
165    /// The underlying AEAD instance. This also does decryption.
166    encryptor: A::AeadImpl,
167    /// The base nonce which we XOR with sequence numbers
168    base_nonce: AeadNonce<A>,
169    /// The exporter secret, used in the `export()` method
170    exporter_secret: ExporterSecret<Kdf>,
171    /// The running sequence number
172    seq: Seq,
173    /// This binds the `AeadCtx` to the KEM that made it. Used to generate `suite_id`.
174    src_kem: PhantomData<Kem>,
175    /// The full ID of the ciphersuite that created this `AeadCtx`. Used for context binding.
176    suite_id: FullSuiteId,
177}

195impl<A: Aead, Kdf: KdfTrait, Kem: KemTrait> AeadCtx<A, Kdf, Kem> {
196    /// Makes an AeadCtx from a raw key and nonce
197    pub(crate) fn new(
198        key: &AeadKey<A>,
199        base_nonce: AeadNonce<A>,
200        exporter_secret: ExporterSecret<Kdf>,
201    ) -> AeadCtx<A, Kdf, Kem> {
202        let suite_id = full_suite_id::<A, Kdf, Kem>();
203        AeadCtx {
204            overflowed: false,
205            encryptor: <A::AeadImpl as aead::KeyInit>::new(&key.0),
206            base_nonce,
207            exporter_secret,
208            seq: <Seq as Default>::default(),
209            src_kem: PhantomData,
210            suite_id,
211        }
212    }
213
214    // RFC 9180 §5.3
215    // def Context.Export(exporter_context, L):
216    //   return LabeledExpand(self.exporter_secret, "sec",
217    //                        exporter_context, L)
218
219    /// Fills a given buffer with secret bytes derived from this encryption context. This value
220    /// does not depend on sequence number, so it is constant for the lifetime of this context.
221    ///
222    /// Return Value
223    /// ============
224    /// Returns `Ok(())` on success. If the buffer length is more than 255x the digest size (in
225    /// bytes) of the underlying hash function, returns an `Err(HpkeError::KdfOutputTooLong)`. Just
226    /// don't use to fill massive buffers and you'll be fine.
227    pub fn export(&self, exporter_ctx: &[u8], out_buf: &mut [u8]) -> Result<(), HpkeError> {
228        // Use our exporter secret as the PRK for an HKDF-Expand op. The only time this fails is
229        // when the length of the PRK is not the the underlying hash function's digest size. But
230        // that's guaranteed by the type system, so we can unwrap().
231        let hkdf_ctx = SimpleHkdf::<Kdf>::from_prk(self.exporter_secret.0.as_slice()).unwrap();
232
233        // This call either succeeds or returns hkdf::InvalidLength (iff the buffer length is more
234        // than 255x the digest size of the underlying hash function)
235        hkdf_ctx
236            .labeled_expand(&self.suite_id, b"sec", exporter_ctx, out_buf)
237            .map_err(|_| HpkeError::KdfOutputTooLong)
238    }
239}

259impl<A: Aead, Kdf: KdfTrait, Kem: KemTrait> AeadCtxR<A, Kdf, Kem> {
260    // RFC 9180 §5.2
261    // def ContextR.Open(aad, ct):
262    //   pt = Open(self.key, self.ComputeNonce(self.seq), aad, ct)
263    //   if pt == OpenError:
264    //     raise OpenError
265    //   self.IncrementSeq()
266    //   return pt
267
268    /// Does a "detached open in place", meaning it overwrites `ciphertext` with the resulting
269    /// plaintext, and takes the tag as a separate input.
270    ///
271    /// Return Value
272    /// ============
273    /// Returns `Ok(())` on success. If this context has been used for so many encryptions that the
274    /// sequence number overflowed, returns `Err(HpkeError::MessageLimitReached)`. If this happens,
275    /// `ciphertext` will be unmodified. If the tag fails to validate, returns
276    /// `Err(HpkeError::OpenError)`. If this happens, `ciphertext` is in an undefined state.
277    pub fn open_in_place_detached(
278        &mut self,
279        ciphertext: &mut [u8],
280        aad: &[u8],
281        tag: &AeadTag<A>,
282    ) -> Result<(), HpkeError> {
283        if self.0.overflowed {
284            // If the sequence counter overflowed, we've been used for too long. Shut down.
285            Err(HpkeError::MessageLimitReached)
286        } else {
287            // Compute the nonce and do the encryption in place
288            let nonce = mix_nonce::<A>(&self.0.base_nonce, &self.0.seq);
289            let decrypt_res = self
290                .0
291                .encryptor
292                .decrypt_in_place_detached(&nonce.0, aad, ciphertext, &tag.0);
293
294            if decrypt_res.is_err() {
295                // Opening failed due to a bad tag
296                return Err(HpkeError::OpenError);
297            }
298
299            // Opening was a success. Try to increment the sequence counter. If it fails, this was
300            // our last decryption.
301            match increment_seq(&self.0.seq) {
302                Some(new_seq) => self.0.seq = new_seq,
303                None => self.0.overflowed = true,
304            }
305
306            Ok(())
307        }
308    }
309
310    /// Opens the given ciphertext and returns a plaintext
311    ///
312    /// Return Value
313    /// ============
314    /// Returns `Ok(())` on success. If this context has been used for so many encryptions that the
315    /// sequence number overflowed, returns `Err(HpkeError::MessageLimitReached)`. If the tag fails
316    /// to validate, returns `Err(HpkeError::OpenError)`.
317    #[cfg_attr(docsrs, doc(cfg(any(feature = "alloc", feature = "std"))))]
318    #[cfg(any(feature = "alloc", feature = "std"))]
319    pub fn open(&mut self, ciphertext: &[u8], aad: &[u8]) -> Result<crate::Vec<u8>, HpkeError> {
320        // Make sure the auth'd ciphertext is long enough to contain a tag. If it isn't, it's
321        // certainly not valid.
322        let tag_len = AeadTag::<A>::size();
323        let msg_len = ciphertext
324            .len()
325            .checked_sub(tag_len)
326            .ok_or(HpkeError::OpenError)?;
327
328        // Now deconstruct the auth'd ciphertext
329        let (ciphertext, tag_slice) = ciphertext.split_at(msg_len);
330        let mut buf = ciphertext.to_vec();
331        let tag = {
332            let mut t = <AeadTag<A> as Default>::default();
333            t.0.copy_from_slice(tag_slice);
334            t
335        };
336
337        // Decrypt and return the decrypted buffer
338        self.open_in_place_detached(&mut buf, aad, &tag)?;
339        Ok(buf)
340    }
341
342    /// Fills a given buffer with secret bytes derived from this encryption context. This value
343    /// does not depend on sequence number, so it is constant for the lifetime of this context.
344    ///
345    /// Return Value
346    /// ============
347    /// Returns `Ok(())` on success. If the buffer length is more than about 255x the digest size
348    /// (in bytes) of the underlying hash function, returns an `Err(HpkeError::KdfOutputTooLong)`.
349    /// The exact number is given in the "Input Length Restrictions" section of the spec. Just
350    /// don't use to fill massive buffers and you'll be fine.
351    pub fn export(&self, info: &[u8], out_buf: &mut [u8]) -> Result<(), HpkeError> {
352        // Pass to AeadCtx
353        self.0.export(info, out_buf)
354    }
355}

375impl<A: Aead, Kdf: KdfTrait, Kem: KemTrait> AeadCtxS<A, Kdf, Kem> {
376    // RFC 9180 §5.2
377    // def ContextS.Seal(aad, pt):
378    //   ct = Seal(self.key, self.ComputeNonce(self.seq), aad, pt)
379    //   self.IncrementSeq()
380    //   return ct
381
382    /// Does a "detached seal in place", meaning it overwrites `plaintext` with the resulting
383    /// ciphertext, and returns the resulting authentication tag
384    ///
385    /// Return Value
386    /// ============
387    /// Returns `Ok(tag)` on success.  If this context has been used for so many encryptions that
388    /// the sequence number overflowed, returns `Err(HpkeError::MessageLimitReached)`. If this
389    /// happens, `plaintext` will be unmodified. If an error happened during encryption, returns
390    /// `Err(HpkeError::SealError)`. If this happens, the contents of `plaintext` is undefined.
391    pub fn seal_in_place_detached(
392        &mut self,
393        plaintext: &mut [u8],
394        aad: &[u8],
395    ) -> Result<AeadTag<A>, HpkeError> {
396        if self.0.overflowed {
397            // If the sequence counter overflowed, we've been used for far too long. Shut down.
398            Err(HpkeError::MessageLimitReached)
399        } else {
400            // Compute the nonce and do the encryption in place
401            let nonce = mix_nonce::<A>(&self.0.base_nonce, &self.0.seq);
402            let tag = self
403                .0
404                .encryptor
405                .encrypt_in_place_detached(&nonce.0, aad, plaintext)
406                .map_err(|_| HpkeError::SealError)?;
407
408            // Try to increment the sequence counter. If it fails, this was our last encryption.
409            match increment_seq(&self.0.seq) {
410                Some(new_seq) => self.0.seq = new_seq,
411                None => self.0.overflowed = true,
412            }
413
414            // Return the tag
415            Ok(AeadTag(tag))
416        }
417    }
418
419    /// Seals the given plaintext and returns the ciphertext
420    ///
421    /// Return Value
422    /// ============
423    /// Returns `Ok(ciphertext)` on success.  If this context has been used for so many encryptions
424    /// that the sequence number overflowed, returns `Err(HpkeError::MessageLimitReached)`. If an
425    /// error happened during encryption, returns `Err(HpkeError::SealError)`.
426    #[cfg_attr(docsrs, doc(cfg(any(feature = "alloc", feature = "std"))))]
427    #[cfg(any(feature = "alloc", feature = "std"))]
428    pub fn seal(&mut self, plaintext: &[u8], aad: &[u8]) -> Result<crate::Vec<u8>, HpkeError> {
429        let msg_len = plaintext.len();
430        let tag_len = AeadTag::<A>::size();
431
432        // Make a buffer that can hold a ciphertext + tag. Copy in the plaintext
433        let mut buf = vec![0u8; msg_len + tag_len];
434        buf[..msg_len].copy_from_slice(plaintext);
435
436        // Seal with a detached tag
437        let tag = self.seal_in_place_detached(&mut buf[..plaintext.len()], aad)?;
438        // Then append the tag to the end of the buffer. The buffer is now the auth'd ciphertext
439        buf[msg_len..msg_len + tag_len].copy_from_slice(&tag.0);
440
441        Ok(buf)
442    }
443
444    /// Fills a given buffer with secret bytes derived from this encryption context. This value
445    /// does not depend on sequence number, so it is constant for the lifetime of this context.
446    ///
447    /// Return Value
448    /// ============
449    /// Returns `Ok(())` on success. If the buffer length is more than 255x the digest size (in
450    /// bytes) of the underlying hash function, returns an `Err(HpkeError::KdfOutputTooLong)`. Just
451    /// don't use to fill massive buffers and you'll be fine.
452    pub fn export(&self, info: &[u8], out_buf: &mut [u8]) -> Result<(), HpkeError> {
453        // Pass to AeadCtx
454        self.0.export(info, out_buf)
455    }
456}

566    macro_rules! test_overflow {
567        ($test_name:ident, $kem_ty:ty) => {
568            #[test]
569            fn $test_name() {
570                type Kem = $kem_ty;
571                type Kdf = HkdfSha256;
572                // Again, this test is cipher-agnostic
573                type A = ChaCha20Poly1305;
574
575                // Make a sequence number that's at the max
576                let big_seq = {
577                    let mut seq = <Seq as Default>::default();
578                    seq.0 = u64::MAX;
579                    seq
580                };
581
582                let (mut sender_ctx, mut receiver_ctx) = gen_ctx_simple_pair::<A, Kdf, Kem>();
583                sender_ctx.0.seq = big_seq.clone();
584                receiver_ctx.0.seq = big_seq.clone();
585
586                // These should support precisely one more encryption before it registers an
587                // overflow
588
589                let msg = b"draxx them sklounst";
590                let aad = b"you have to have the kebapi";
591
592                // Do one round trip and ensure it works
593                {
594                    let mut buf = msg.clone();
595
596                    // Encrypt the plaintext
597                    let ciphertext = sender_ctx.seal(&mut buf, aad).expect("seal() failed");
598
599                    // Now to decrypt on the other side
600                    let roundtrip_plaintext =
601                        receiver_ctx.open(&ciphertext, aad).expect("open() failed");
602
603                    // Make sure the output message was the same as the input message
604                    assert_eq!(msg, roundtrip_plaintext.as_slice());
605                }
606
607                // Try another round trip and ensure that we've overflowed
608                {
609                    // Try to encrypt the plaintext
610                    match sender_ctx.seal(msg, aad) {
611                        Err(HpkeError::MessageLimitReached) => {
612                            // Good, this should have overflowed
613                        }
614                        Err(e) => panic!("seal() should have overflowed. Instead got {}", e),
615                        _ => panic!("seal() should have overflowed. Instead it succeeded"),
616                    }
617
618                    // Now try to decrypt something. This isn't a valid ciphertext or tag, but the
619                    // overflow should fail before the tag check fails.
620                    let placeholder_ciphertext = [0u8; 32];
621
622                    match receiver_ctx.open(&placeholder_ciphertext, aad) {
623                        Err(HpkeError::MessageLimitReached) => {
624                            // Good, this should have overflowed
625                        }
626                        Err(e) => panic!("open() should have overflowed. Instead got {}", e),
627                        _ => panic!("open() should have overflowed. Instead it succeeded"),
628                    }
629                }
630            }
631        };
632    }

634    /// Tests that `open()` can decrypt things properly encrypted with `seal()`
635    #[cfg(any(feature = "alloc", feature = "std"))]
636    macro_rules! test_ctx_correctness {
637        ($test_name:ident, $aead_ty:ty, $kem_ty:ty) => {
638            #[test]
639            fn $test_name() {
640                type A = $aead_ty;
641                type Kdf = HkdfSha256;
642                type Kem = $kem_ty;
643
644                let (mut sender_ctx, mut receiver_ctx) = gen_ctx_simple_pair::<A, Kdf, Kem>();
645
646                let msg = b"Love it or leave it, you better gain way";
647                let aad = b"You better hit bull's eye, the kid don't play";
648
649                // Encrypt in place with the sender context
650                let ciphertext = sender_ctx.seal(msg, aad).expect("seal() failed");
651
652                // Make sure seal() isn't a no-op
653                assert_ne!(&ciphertext, msg);
654
655                // Decrypt with the receiver context
656                let decrypted = receiver_ctx.open(&ciphertext, aad).expect("open() failed");
657                assert_eq!(&decrypted, msg);
658
659                // Now try sending an invalid message followed by a valid message. The valid
660                // message should decrypt correctly
661                let invalid_ciphertext = [0x00; 32];
662                assert!(receiver_ctx.open(&invalid_ciphertext, aad).is_err());
663
664                // Now make sure a round trip succeeds
665                let ciphertext = sender_ctx.seal(msg, aad).expect("second seal() failed");
666
667                // Decrypt with the receiver context
668                let decrypted = receiver_ctx
669                    .open(&ciphertext, aad)
670                    .expect("second open() failed");
671                assert_eq!(&decrypted, msg);
672            }
673        };
674    }
