31/// Represents key derivation functionality
32pub trait Kdf: Sized {
33    /// The algorithm identifier for a KDF implementation
34    #[doc(hidden)]
35    const KDF_ID: u16;
36
37    /// The security strength of the KDF, in bytes
38    #[doc(hidden)]
39    type Nh: ArraySize;
40
41    /// Combines the context of the given mode and info string into the shared secret. Produces an
42    /// encryption context using the resulting key material.
43    ///
44    /// # Panics
45    /// Panics if `info.len() + mode.get_psk_id().len() + 5` ≥ 2¹⁶
46    #[doc(hidden)]
47    fn combine_secrets<A, Kem, O>(
48        mode: &O,
49        shared_secret: SharedSecret<Kem>,
50        info: &[u8],
51    ) -> AeadCtx<A, Self, Kem>
52    where
53        A: Aead,
54        Kem: KemTrait,
55        O: OpMode<Kem>;
56
57    /// Extracts randomness from `ikm`, binds it to the given suite ID and info string, and expands
58    /// the result to fill the output buffer.  If `out.len()` is more than 255x the digest size (in
59    /// bytes) of the underlying hash function, returns an `Err(HpkeError::KdfOutputTooLong)`.
60    #[doc(hidden)]
61    fn extract_and_expand(
62        ikm: &[u8],
63        suite_id: &[u8],
64        info: &[u8],
65        out: &mut [u8],
66    ) -> Result<(), HpkeError>;
67
68    /// Derives bytes for a use as a P256/P384/P521 ephemeral secret. Counter is because it may
69    /// require multiple attempts to find a valid scalar. The keying material SHOULD have as many
70    /// bits of entropy as the bit length of a secret key.
71    #[doc(hidden)]
72    fn derive_nistp_sk_eph_bytes<PrivateKeySize: ArraySize>(
73        suite_id: &KemSuiteId,
74        ikm: &[u8],
75        counter: u8,
76    ) -> Array<u8, PrivateKeySize>;
77
78    /// Derives bytes for a use as an x25519 ephemeral secret. There is no counter because no
79    /// retries are necessary; every output is a valid secret key. The keying material SHOULD have
80    /// as many bits of entropy as the bit length of a secret key, i.e., 256.
81    #[doc(hidden)]
82    fn derive_x25519_sk_eph_bytes(suite_id: &KemSuiteId, ikm: &[u8]) -> [u8; 32];
83
84    /// Constructs the export secret of an encryption context. Returns
85    /// `Err(HpkeError::KdfOutputTooLong)` if `out_buf.len()` ≥ 2¹⁶.
86    #[doc(hidden)]
87    fn export(
88        exporter_secret: &[u8],
89        suite_id: &[u8],
90        exporter_ctx: &[u8],
91        out_buf: &mut [u8],
92    ) -> Result<(), HpkeError>;
93}

115#[cfg(feature = "hkdfsha2")]
116impl KdfTrait for HkdfSha256 {
117    // RFC 9180 §7.2: HKDF-SHA256
118    const KDF_ID: u16 = 0x0001;
119    type Nh = U32;
120
121    fn combine_secrets<A, Kem, O>(
122        mode: &O,
123        shared_secret: SharedSecret<Kem>,
124        info: &[u8],
125    ) -> AeadCtx<A, Self, Kem>
126    where
127        A: Aead,
128        Kem: KemTrait,
129        O: OpMode<Kem>,
130    {
131        two_stage_kdf::combine_secrets::<_, Sha256, _, _, _>(mode, shared_secret, info)
132    }
133
134    fn extract_and_expand(
135        ikm: &[u8],
136        suite_id: &[u8],
137        info: &[u8],
138        out: &mut [u8],
139    ) -> Result<(), HpkeError> {
140        two_stage_kdf::extract_and_expand::<Sha256>(ikm, suite_id, info, out)
141    }
142
143    fn derive_x25519_sk_eph_bytes(suite_id: &KemSuiteId, ikm: &[u8]) -> [u8; 32] {
144        two_stage_kdf::derive_x25519_sk_eph_bytes::<Sha256>(suite_id, ikm)
145    }
146
147    fn derive_nistp_sk_eph_bytes<PrivateKeySize: ArraySize>(
148        suite_id: &KemSuiteId,
149        ikm: &[u8],
150        counter: u8,
151    ) -> Array<u8, PrivateKeySize> {
152        two_stage_kdf::derive_nistp_sk_eph_bytes::<Sha256, PrivateKeySize>(suite_id, ikm, counter)
153    }
154
155    fn export(
156        exporter_secret: &[u8],
157        suite_id: &[u8],
158        exporter_ctx: &[u8],
159        out_buf: &mut [u8],
160    ) -> Result<(), HpkeError> {
161        two_stage_kdf::export::<Sha256>(exporter_secret, suite_id, exporter_ctx, out_buf)
162    }
163}
