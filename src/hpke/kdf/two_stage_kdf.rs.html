17// RFC 9180 §4.1
18// def ExtractAndExpand(dh, kem_context):
19//   eae_prk = LabeledExtract("", "eae_prk", dh)
20//   shared_secret = LabeledExpand(eae_prk, "shared_secret",
21//                                 kem_context, Nsecret)
22//   return shared_secret
23
24/// Uses the given IKM to extract a secret, and then uses that secret, plus the given suite ID and
25/// info string, to expand to the output buffer. Uses HKDF rather than XOF.
26///
27/// If `out.len()` is more than 255x the digest size (in bytes) of the underlying hash function,
28/// returns an `Err(HpkeError::KdfOutputTooLong)`.
29pub(crate) fn extract_and_expand<H>(
30    ikm: &[u8],
31    suite_id: &[u8],
32    info: &[u8],
33    out: &mut [u8],
34) -> Result<(), HpkeError>
35where
36    H: Clone + Digest + EagerHash,
37{
38    // Extract using given IKM
39    let (_, hkdf_ctx) = labeled_extract::<H>(&[], suite_id, b"eae_prk", ikm);
40    // Expand using given info string
41    labeled_expand(&hkdf_ctx, suite_id, b"shared_secret", info, out)
42        .map_err(|hkdf::InvalidLength| HpkeError::KdfOutputTooLong)
43}

45// RFC 9180 §4
46// def LabeledExtract(salt, label, ikm):
47//   labeled_ikm = concat("HPKE-v1", suite_id, label, ikm)
48//   return Extract(salt, labeled_ikm)
49
50/// Returns the HKDF context derived from `(salt=salt, ikm="HPKE-v1"||suite_id||label||ikm)`
51fn labeled_extract<H>(
52    salt: &[u8],
53    suite_id: &[u8],
54    label: &[u8],
55    ikm: &[u8],
56) -> (
57    Array<u8, <<H as EagerHash>::Core as OutputSizeUser>::OutputSize>,
58    Hkdf<H>,
59)
60where
61    H: Clone + Digest + EagerHash,
62{
63    // Call HKDF-Extract with the IKM being the concatenation of all of the above
64    let mut extract_ctx = HkdfExtract::<H>::new(Some(salt));
65    extract_ctx.input_ikm(VERSION_LABEL);
66    extract_ctx.input_ikm(suite_id);
67    extract_ctx.input_ikm(label);
68    extract_ctx.input_ikm(ikm);
69    extract_ctx.finalize()
70}

72// RFC 9180 §4
73// def LabeledExpand(prk, label, info, L):
74//   labeled_info = concat(I2OSP(L, 2), "HPKE-v1", suite_id,
75//                         label, info)
76//   return Expand(prk, labeled_info, L)
77
78/// Does a `LabeledExpand` key derivation function using HKDF. If `out.len()` is more than 255x the
79/// digest size (in bytes) of the underlying hash function, returns an `Err(hkdf::InvalidLength)`.
80fn labeled_expand<D: Clone + EagerHash>(
81    hkdf_ctx: &Hkdf<D>,
82    suite_id: &[u8],
83    label: &[u8],
84    info: &[u8],
85    out: &mut [u8],
86) -> Result<(), hkdf::InvalidLength> {
87    // We need to write the length as a u16, so that's the de-facto upper bound on length
88    if out.len() > u16::MAX as usize {
89        // The error condition is met, since 2^16 is way bigger than 255 * digest_bytelen
90        return Err(hkdf::InvalidLength);
91    }
92
93    // Encode the output length in the info string
94    let mut len_buf = [0u8; 2];
95    write_u16_be(&mut len_buf, out.len() as u16);
96
97    // Call HKDF-Expand() with the info string set to the concatenation of all of the above
98    let labeled_info = [&len_buf, VERSION_LABEL, suite_id, label, info];
99    hkdf_ctx.expand_multi_info(&labeled_info, out)
100}

102// This is the KeySchedule function. It runs a KDF over all the parameters, inputs, and secrets,
103// and spits out a key-nonce pair to be used for symmetric encryption.
104pub(crate) fn combine_secrets<A, H, Kdf, Kem, O>(
105    mode: &O,
106    shared_secret: SharedSecret<Kem>,
107    info: &[u8],
108) -> AeadCtx<A, Kdf, Kem>
109where
110    A: Aead,
111    H: Clone + Digest + EagerHash,
112    Kdf: KdfTrait,
113    Kem: KemTrait,
114    O: OpMode<Kem>,
115{
116    // Put together the binding context used for all KDF operations
117    let suite_id = full_suite_id::<A, Kdf, Kem>();
118
119    // In KeySchedule() in RFC 9180,
120    //   psk_id_hash = LabeledExtract("", "psk_id_hash", psk_id)
121    //   info_hash = LabeledExtract("", "info_hash", info)
122    //   key_schedule_context = concat(mode, psk_id_hash, info_hash)
123
124    // We concat without allocation by making a buffer of the maximum possible size, then
125    // taking the appropriately sized slice.
126    let (sched_context_buf, sched_context_size) = {
127        let (psk_id_hash, _) =
128            labeled_extract::<H>(&[], &suite_id, b"psk_id_hash", mode.get_psk_id());
129        let (info_hash, _) = labeled_extract::<H>(&[], &suite_id, b"info_hash", info);
130
131        // Yes it's overkill to bound the first input by MAX_DIGEST_SIZE, since it's only 1 byte.
132        // But whatever, this is pretty clean.
133        concat_with_known_maxlen!(
134            MAX_DIGEST_SIZE,
135            &[mode.mode_id()],
136            psk_id_hash.as_slice(),
137            info_hash.as_slice()
138        )
139    };
140    let sched_context = &sched_context_buf[..sched_context_size];
141
142    // In KeySchedule(),
143    //   secret = LabeledExtract(shared_secret, "secret", psk)
144    //   key = LabeledExpand(secret, "key", key_schedule_context, Nk)
145    //   base_nonce = LabeledExpand(secret, "base_nonce", key_schedule_context, Nn)
146    //   exporter_secret = LabeledExpand(secret, "exp", key_schedule_context, Nh)
147    // Instead of `secret` we derive an HKDF context which we run .expand() on to derive the
148    // key-nonce pair.
149    let (_, secret_ctx) =
150        labeled_extract::<H>(&shared_secret.0, &suite_id, b"secret", mode.get_psk_bytes());
151
152    // Empty fixed-size buffers
153    let mut key = crate::aead::AeadKey::<A>::default();
154    let mut base_nonce = crate::aead::AeadNonce::<A>::default();
155    let mut exporter_secret = <ExporterSecret<Kdf> as Default>::default();
156
157    // Fill the key, base nonce, and exporter secret. This only errors if the output values are
158    // 255x the digest size of the hash function. Since these values are fixed at compile time, we
159    // don't worry about it.
160    labeled_expand(
161        &secret_ctx,
162        &suite_id,
163        b"key",
164        sched_context,
165        key.0.as_mut_slice(),
166    )
167    .expect("aead key len is way too big");
168    labeled_expand(
169        &secret_ctx,
170        &suite_id,
171        b"base_nonce",
172        sched_context,
173        base_nonce.0.as_mut_slice(),
174    )
175    .expect("nonce len is way too big");
176    labeled_expand(
177        &secret_ctx,
178        &suite_id,
179        b"exp",
180        sched_context,
181        exporter_secret.0.as_mut_slice(),
182    )
183    .expect("exporter secret len is way too big");
184
185    AeadCtx::new(&key, base_nonce, exporter_secret)
186}

190//   dkp_prk = LabeledExtract("", "dkp_prk", ikm)
191//   sk = LabeledExpand(dkp_prk, "sk", "", Nsk)

194/// Derive secret key bytes for x25519 using a two-stage KDF
195pub(crate) fn derive_x25519_sk_eph_bytes<H>(suite_id: &KemSuiteId, ikm: &[u8]) -> [u8; 32]
196where
197    H: Clone + Digest + EagerHash,
198{
199    // Write the label into a byte buffer and extract from the IKM
200    let (_, hkdf_ctx) = labeled_extract::<H>(&[], suite_id, b"dkp_prk", ikm);
201    // The buffer we hold the candidate scalar bytes in. This is the size of a private key.
202    let mut buf = [0u8; 32];
203    labeled_expand(&hkdf_ctx, suite_id, b"sk", &[], &mut buf).unwrap();
204
205    buf
206}

208// RFC 9180 §7.1.3:
209// def DeriveKeyPair(ikm):
210//   dkp_prk = LabeledExtract("", "dkp_prk", ikm)
211//   sk = 0
212//   counter = 0
213//   while sk == 0 or sk >= order:
214//     if counter > 255:
215//       raise DeriveKeyPairError
216//     bytes = LabeledExpand(dkp_prk, "candidate",
217//                           I2OSP(counter, 1), Nsk)
218//     bytes[0] = bytes[0] & bitmask
219//     sk = OS2IP(bytes)
220//     counter = counter + 1
221//   return (sk, pk(sk))
222// where `bitmask` is defined to be 0xFF for P-256 and P-384, and 0x01 for P-521
223
224/// Derive candidate secret key bytes for p256/p384/p521 using a two-stage KDF
225pub(crate) fn derive_nistp_sk_eph_bytes<H, PrivateKeySize>(
226    suite_id: &KemSuiteId,
227    ikm: &[u8],
228    counter: u8,
229) -> Array<u8, PrivateKeySize>
230where
231    H: Clone + Digest + EagerHash,
232    PrivateKeySize: ArraySize,
233{
234    // Write the label into a byte buffer and extract from the IKM
235    let (_, hkdf_ctx) = labeled_extract::<H>(&[], suite_id, b"dkp_prk", ikm);
236
237    // The buffer we hold the candidate scalar bytes in. This is the size of a
238    // private key.
239    let mut buf = Array::<u8, PrivateKeySize>::default();
240
241    // This unwrap is fine. It only triggers if buf is way too big. It's only
242    // 32 bytes.
243    labeled_expand(&hkdf_ctx, suite_id, b"candidate", &[counter], &mut buf).unwrap();
244
245    buf
246}

248// RFC 9180 §5.3
249// def Context.Export(exporter_context, L):
250//   return LabeledExpand(self.exporter_secret, "sec",
251//                        exporter_context, L)
252
253/// Derive an exporter secret using a two-stage KDF. Returns `Err(HpkeError::KdfOutputTooLong)` if
254/// `out_buf.len()` ≥ 2¹⁶.
255pub(crate) fn export<H>(
256    exporter_secret: &[u8],
257    suite_id: &[u8],
258    exporter_ctx: &[u8],
259    out_buf: &mut [u8],
260) -> Result<(), HpkeError>
261where
262    H: Clone + Digest + EagerHash,
263{
264    // Use our exporter secret as the PRK for an HKDF-Expand op. The only time this fails is
265    // when the length of the PRK is not the the underlying hash function's digest size. But
266    // that's guaranteed by the type system, so we can unwrap().
267    let hkdf_ctx = Hkdf::<H>::from_prk(exporter_secret).unwrap();
268
269    // This call either succeeds or returns hkdf::InvalidLength (iff the buffer length is more
270    // than 255x the digest size of the underlying hash function)
271    labeled_expand(&hkdf_ctx, suite_id, b"sec", exporter_ctx, out_buf)
272        .map_err(|_| HpkeError::KdfOutputTooLong)
