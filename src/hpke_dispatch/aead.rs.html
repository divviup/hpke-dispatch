6Aead represents an authenticated encryption with additional data
7encryption function, as per [RFC9180ยง7.3](https://www.rfc-editor.org/rfc/rfc9180.html#section-7.3)

9#[non_exhaustive]

17pub enum Aead {
18    #[cfg(feature = "aead-aes-gcm-128")]
19    /// AES-128-GCM [GCM](https://doi.org/10.6028/nist.sp.800-38d)
20    AesGcm128 = 1,
21    #[cfg(feature = "aead-aes-gcm-256")]
22    /// AES-256-GCM [GCM](https://doi.org/10.6028/nist.sp.800-38d)
23    AesGcm256 = 2,
24    #[cfg(feature = "aead-chacha-20-poly-1305")]
25    /// ChaCha20Poly1305 [RFC8439](https://www.rfc-editor.org/info/rfc8439)
26    ChaCha20Poly1305 = 3,
27}
28
29impl FromStr for Aead {
30    type Err = IdLookupError;
31
32    fn from_str(s: &str) -> Result<Self, Self::Err> {
33        match &*s.to_lowercase().replace('-', "") {
34            #[cfg(feature = "aead-aes-gcm-128")]
35            "aesgcm128" | "aes128gcm" => Ok(Self::AesGcm128),
36            #[cfg(feature = "aead-aes-gcm-256")]
37            "aesgcm256" | "aes256gcm" => Ok(Self::AesGcm256),
38            #[cfg(feature = "aead-chacha-20-poly-1305")]
39            "chacha20poly1305" => Ok(Self::ChaCha20Poly1305),
40            _ => Err(IdLookupError("aead not recognized")),
41        }
42    }
43}
