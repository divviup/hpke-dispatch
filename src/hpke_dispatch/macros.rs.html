3macro_rules! match_algo {
4    ($aead:ident, $kdf:ident, $kem:ident, $fn:ident) => {
5        match_algo!(@aead, $aead, $kdf, $kem, $fn)
6    };
7
8    (@aead, $aead:ident, $kdf:ident, $kem:ident, $fn:ident) => {
9        match $aead {
10            #[cfg(feature = "aead-aes-gcm-128")]
11            $crate::Aead::AesGcm128 => match_algo!(@kdf, hpke::aead::AesGcm128, $kdf, $kem, $fn),
12            #[cfg(feature = "aead-aes-gcm-256")]
13            $crate::Aead::AesGcm256 => match_algo!(@kdf, hpke::aead::AesGcm256, $kdf, $kem, $fn),
14            #[cfg(feature = "aead-chacha-20-poly-1305")]
15            $crate::Aead::ChaCha20Poly1305 => {
16                match_algo!(@kdf, hpke::aead::ChaCha20Poly1305, $kdf, $kem, $fn)
17            }
18        }
19    };
20
21    (@kdf, $aead:ty, $kdf:expr, $kem:expr, $fn:ident) => {
22        match $kdf {
23            #[cfg(feature = "kdf-sha256")]
24            $crate::Kdf::Sha256 => match_algo!(@kem, $aead, hpke::kdf::HkdfSha256, $kem, $fn),
25            #[cfg(feature = "kdf-sha384")]
26            $crate::Kdf::Sha384 => match_algo!(@kem, $aead, hpke::kdf::HkdfSha384, $kem, $fn),
27            #[cfg(feature = "kdf-sha512")]
28            $crate::Kdf::Sha512 => match_algo!(@kem, $aead, hpke::kdf::HkdfSha512, $kem, $fn),
29        }
30    };
31
32    (@kem, $aead:ty, $kdf:ty, $kem:expr, $fn:ident) => {
33        match $kem {
34            #[cfg(feature = "kem-dh-p256-hkdf-sha256")]
35            $crate::Kem::DhP256HkdfSha256 => $fn::<$aead, $kdf, hpke::kem::DhP256HkdfSha256>,
36            #[cfg(feature = "kem-dh-p384-hkdf-sha384")]
37            $crate::Kem::DhP384HkdfSha384 => $fn::<$aead, $kdf, hpke::kem::DhP384HkdfSha384>,
38            #[cfg(feature = "kem-dh-p521-hkdf-sha512")]
39            $crate::Kem::DhP521HkdfSha512 => $fn::<$aead, $kdf, hpke::kem::DhP521HkdfSha512>,
40            #[cfg(feature = "kem-x25519-hkdf-sha256")]
41            $crate::Kem::X25519HkdfSha256 => $fn::<$aead, $kdf, hpke::kem::X25519HkdfSha256>,
42        }
43    };
44}
45
46pub(crate) use match_algo;
