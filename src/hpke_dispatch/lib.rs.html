66/**
67A simple error type for failed id lookups
68 */
69#[derive(Copy, Clone, Debug)]
70#[cfg_attr(
71    feature = "serde",
72    derive(serde_crate::Serialize, serde_crate::Deserialize)
73)]
74#[cfg_attr(feature = "serde", serde(crate = "serde_crate"))]
75pub struct IdLookupError(&'static str);
76impl std::fmt::Display for IdLookupError {
77    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
78        f.write_fmt(format_args!("id lookup error {}", self.0))
79    }
80}
81impl std::error::Error for IdLookupError {}
