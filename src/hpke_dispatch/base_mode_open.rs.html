5/// Single-shot HPKE ciphertext opening.
6///
7/// `base_mode_open` provides an interface to [`hpke::single_shot_open`] that does not require
8/// compile time selection of an algorithm. Instead, the selected algorithm is provided through the
9/// [`Config`] passed as the first argument.
10///
11/// # Errors
12///
13/// This will return a `Result::Err` variant if:
14///
15/// * we are unable to deserialize the private key or encapsulated key
16/// * there is an error in key decapsulation
17/// * there is an error in decryption
18pub fn base_mode_open(
19    config: &Config,
20    private_key: &[u8],
21    encapped_key: &[u8],
22    info: &[u8],
23    ciphertext: &[u8],
24    aad: &[u8],
25) -> Result<Vec<u8>, HpkeError> {
26    let Config { aead, kdf, kem } = *config;
27    let open = match_algo!(aead, kdf, kem, open);
28    open(private_key, encapped_key, info, ciphertext, aad)
29}
