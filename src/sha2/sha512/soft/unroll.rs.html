29/// Read round constant
30fn rk(i: usize) -> u64 {
31    // `read_volatile` forces the compiler to read round constants from the static
32    // instead of inlining them, which improves codegen and performance
33    unsafe {
34        let p = K64.as_ptr().add(i);
35        core::ptr::read_volatile(p)
36    }
37}

39/// Process a block with the SHA-512 algorithm.
40fn compress_block(state: &mut [u64; 8], block: &[u8; 128]) {
41    let mut block = super::to_u64s(block);
42    let [mut a, mut b, mut c, mut d, mut e, mut f, mut g, mut h] = *state;
43
44    repeat80!(i, {
45        let w = if i < 16 {
46            block[i]
47        } else {
48            let w15 = block[(i - 15) % 16];
49            let s0 = (w15.rotate_right(1)) ^ (w15.rotate_right(8)) ^ (w15 >> 7);
50            let w2 = block[(i - 2) % 16];
51            let s1 = (w2.rotate_right(19)) ^ (w2.rotate_right(61)) ^ (w2 >> 6);
52            block[i % 16] = block[i % 16]
53                .wrapping_add(s0)
54                .wrapping_add(block[(i - 7) % 16])
55                .wrapping_add(s1);
56            block[i % 16]
57        };
58
59        let s1 = e.rotate_right(14) ^ e.rotate_right(18) ^ e.rotate_right(41);
60        let ch = (e & f) ^ ((!e) & g);
61        let t1 = s1
62            .wrapping_add(ch)
63            .wrapping_add(rk(i))
64            .wrapping_add(w)
65            .wrapping_add(h);
66        let s0 = a.rotate_right(28) ^ a.rotate_right(34) ^ a.rotate_right(39);
67        let maj = (a & b) ^ (a & c) ^ (b & c);
68        let t2 = s0.wrapping_add(maj);
69
70        h = g;
71        g = f;
72        f = e;
73        e = d.wrapping_add(t1);
74        d = c;
75        c = b;
76        b = a;
77        a = t1.wrapping_add(t2);
78    });
79
80    state[0] = state[0].wrapping_add(a);
81    state[1] = state[1].wrapping_add(b);
82    state[2] = state[2].wrapping_add(c);
83    state[3] = state[3].wrapping_add(d);
84    state[4] = state[4].wrapping_add(e);
85    state[5] = state[5].wrapping_add(f);
86    state[6] = state[6].wrapping_add(g);
87    state[7] = state[7].wrapping_add(h);
88}

90pub(in super::super) fn compress(state: &mut [u64; 8], blocks: &[[u8; 128]]) {
91    for block in blocks {
92        compress_block(state, block);
93    }
